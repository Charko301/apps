🔗 Google連携（GAS）
Google Apps Script のデプロイURLを設定すると、
・次回予約 → Googleカレンダーに自動登録
・受診記録 → スプレッドシートに自動書き出し
が可能になります。
📅 次回予約をカレンダーに自動登録 受診記録保存時に実行
📊 受診記録をシートに自動同期 受診記録保存時に実行
💉 デュピクセント記録をシートに同期 注射記録保存時に実行
① script.google.com で新規プロジェクト作成
② 下記コードを貼り付け、スプレッドシートIDとカレンダーIDを設定
③「デプロイ」→「新しいデプロイ」→ Webアプリ・全ユーザーアクセス可 → URLをコピー
// ===== GAS テンプレート ===== // ↓ 各自の設定に変更してください const SHEET_ID = 'スプレッドシートのID'; const CAL_ID = 'Googleカレンダーの ID（例：primary）'; const SHEET_VISIT = '受診記録'; const SHEET_DUP = 'デュピクセント'; function doPost(e) { try { const body = JSON.parse(e.postData.contents); const action = body.action; if (action === 'addCalendar') { addCalendarEvent(body); } else if (action === 'addVisit') { addVisitRow(body); } else if (action === 'addDupixent') { addDupRow(body); } return ContentService .createTextOutput(JSON.stringify({ok: true})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({ok: false, error: err.message})) .setMimeType(ContentService.MimeType.JSON); } } function addCalendarEvent(b) { const cal = CalendarApp.getCalendarById(CAL_ID); const startStr = b.date + (b.time ? 'T' + b.time + ':00' : 'T09:00:00'); const start = new Date(startStr); const end = new Date(start.getTime() + 60 * 60 * 1000); cal.createEvent(b.title, start, end, {description: b.description || ''}); } function addVisitRow(b) { const ss = SpreadsheetApp.openById(SHEET_ID); let sh = ss.getSheetByName(SHEET_VISIT); if (!sh) { sh = ss.insertSheet(SHEET_VISIT); sh.appendRow(['日付','時刻','医療機関','医師','種別', '伝えたこと','言われたこと','処方変更','処方メモ', '検査値','次回日','診察費','薬代','支払']); } sh.appendRow([b.date, b.time||'', b.clinic||'', b.doctor||'', b.type||'', b.told||'', b.heard||'', b.rx||'', b.rxMemo||'', b.lab||'', b.nextDate||'', b.cost||0, b.drugCost||0, b.payment||'']); } function addDupRow(b) { const ss = SpreadsheetApp.openById(SHEET_ID); let sh = ss.getSheetByName(SHEET_DUP); if (!sh) { sh = ss.insertSheet(SHEET_DUP); sh.appendRow(['注射日','次回予定日','メモ']); } sh.appendRow([b.date, b.nextDate, b.memo||'']); }
